# frozen_string_literal: true

def make_ngram(n, arr)
  print "#{n}-gram: "
  vector = Hash.new { |h, k| h[k] = 0 }
  arr.each_cons(n).each_with_object(vector) { |c, v| v[c.join] += 1 }.sort_by { |a| a[0] }
end

def print_ngram(ngram)
  puts "[" + ngram.map { |e| e.join(":") }.join(", ") + "]"
end
